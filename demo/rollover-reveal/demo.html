//default
$(".box.slideup").rolloverReveal({
    direction : "up",
    reverse : false,
    overlaySelector: ".overlay",
    openDuration: 300,
    closeDuration: 300,
    openEasing: 'swing',
    closeEasing: 'swing'
});

$(".box.slideright").rolloverReveal({
    direction : "right",
    closeEasing: "easeOutBounce",
    closeDuration : 800
});
$(".box.slidedown").rolloverReveal({
    direction : "down"
});
$(".box.slideleft").rolloverReveal({
    direction : "left",
    reverse: true
});
